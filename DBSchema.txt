## Database schema

CREATE TABLE ciquiz (
 id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
 question VARCHAR(50) NOT NULL,
 answer BOOLEAN DEFAULT NULL,
 description VARCHAR(140) NOT NULL,
 ) 

CREATE TABLE ciquizimages (
 id INT(6) UNSIGNED AUTO_INCREMENT PRIMARY KEY,
 image LONGBLOB NOT NULL,
)
